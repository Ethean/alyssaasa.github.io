public void BFS(int[][] graph, int[] visit, int node) {
    visit[node] = 1;
    Queue<Integer> que = new LinkedList();
    que.add(node);
    while(!que.isEmpty()) {
        int new_node = que.poll();
        // è®¿é—®nodeèŠ‚ç‚¹è¦æ‰§è¡Œçš„æ“ä½œ

            }
        }
    }
